                    <div class="mb-4 bg-[#0b1329] p-4 rounded-lg border border-gray-800">

                    <h3 class="font-bold text-white mb-2 flex items-center gap-2">1️⃣ The Biological Peak</h3>
                    <p class="text-gray-400 text-xs leading-relaxed">
                        Puncak bagi lengkung derivatif $f'(x)$ (ADG) berlaku pada **titik infleksi** lengkung berat kumulatif $f(x)$. Dari segi matematik, ini adalah titik di mana terbitan kedua bersamaan dengan sifar: $f''(x) = 0$. Secara praktikalnya, ini menandakan kecekapan pertumbuhan biologi maksima bagi lembu potong. Selepas hari ini, kadar pengumpulan berat mula menurun, dan tenaga makanan harian lebih banyak disalurkan untuk penyelenggaraan tubuh berbanding pembentukan tisu otot baharu.
                    </p>
                </div>
                <div>
                    <h3 class="font-bold text-white mb-2 flex items-center gap-2">2️⃣ The Microclimate Equation</h3>
                    <p class="text-gray-400 text-xs leading-relaxed">
                        Di kawasan tropika Malaysia, suhu persekitaran fidlot kerap melebihi 30°C. Di dalam simulator, **Indeks Kecekapan Persekitaran ($\eta$)** mewakili parameter mikroklimat ini. Dalam sistem perumahan terbuka (open house), lembu sering mengalami tekanan haba (heat stress) yang memperlahankan proses enzim metabolik, mengurangkan pengambilan makanan, seterusnya anjakan titik puncak pertumbuhan ke kanan dan merendahkan nilai ADG maksima ($\eta \approx 0.80$). Tetapan $\eta = 1.05$ mencerminkan pengurusan sistem kandang berbumbung dengan kipas atau ventilasi optimum yang mengurangkan tekanan haba.
                    </p>

Dengan merujuk silang kos makanan harian dengan nilai penambahan berat daging kumulatif melalui lengkung derivatif, penternak boleh mengesan dengan tepat tarikh penuaian paling ekonomik sebelum FCR jatuh mendatar, sekaligus memaksimumkan margin keuntungan bersih operasi fidlot.

    <script>
        const beta = 12; 
        
        const daySlider = document.getElementById('daySlider');
        const dayValue = document.getElementById('dayValue');
        const weightVal = document.getElementById('weightVal');
        const adgVal = document.getElementById('adgVal');
        const phaseVal = document.getElementById('phaseVal');
        const laymanText = document.getElementById('laymanText');

        const wmaxSlider = document.getElementById('wmaxSlider');
        const kSlider = document.getElementById('kSlider');
        const etaSlider = document.getElementById('etaSlider');
        const wmaxLabel = document.getElementById('wmaxLabel');
        const kLabel = document.getElementById('kLabel');
        const etaLabel = document.getElementById('etaLabel');

        let chart;

        function calculateGrowth(x, W_max, k, eta) {
            const expTerm = Math.exp(-k * eta * x);
            return W_max / (1 + beta * expTerm);
        }

        function calculateDerivative(x, W_max, k, eta) {
            const r = k * eta;
            const expTerm = Math.exp(-r * x);
            const numerator = W_max * beta * r * expTerm;
            const denominator = Math.pow(1 + beta * expTerm, 2);
            return numerator / denominator;
        }

        function generateChartData() {
            const days = [];
            const weights = [];
            const adgs = [];
            
            const W_max = parseFloat(wmaxSlider.value);
            const k = parseFloat(kSlider.value);
            const eta = parseFloat(etaSlider.value);

            for (let d = 0; d <= 180; d += 5) {
                days.push(d);
                weights.push(calculateGrowth(d, W_max, k, eta).toFixed(1));
                adgs.push((calculateDerivative(d, W_max, k, eta) * 10).toFixed(2)); 
            }
            return { days, weights, adgs };
        }

        function updateMetrics() {
            const x = parseInt(daySlider.value);
            const W_max = parseFloat(wmaxSlider.value);
            const k = parseFloat(kSlider.value);
            const eta = parseFloat(etaSlider.value);

            dayValue.innerText = x;
            wmaxLabel.innerText = W_max;
            kLabel.innerText = k.toFixed(3);
            etaLabel.innerText = eta.toFixed(2);

            const f_x = calculateGrowth(x, W_max, k, eta);
            const fPrime_x = calculateDerivative(x, W_max, k, eta);

            weightVal.innerText = f_x.toFixed(1);
            adgVal.innerText = fPrime_x.toFixed(2);

            const r = k * eta;
            const peakDay = Math.round(Math.log(beta) / r);

            if (x < peakDay - 20) {
                phaseVal.innerText = "Starter / Adaptation";
                laymanText.innerHTML = `Lembu berada di fasa adaptasi fidlot. Kadar tumbesaran (ADG) baru meningkat secara eksponen ke <strong>${fPrime_x.toFixed(2)} kg/hari</strong>. Fokuskan makanan jenis 'starter' tinggi protein kasar untuk menyokong perkembangan tisu otot awal. Pastikan pengudaraan rumah ternakan bersih.`;
            } else if (x >= peakDay - 20 && x <= peakDay + 20) {
                phaseVal.innerText = "Peak Growth Phase";
                laymanText.innerHTML = `<strong>Kadar pertumbuhan maksima dikesan!</strong> Lembu membesar pada kelajuan puncak sekitar <strong>${fPrime_x.toFixed(2)} kg/hari</strong>. Ini adalah titik infleksi matematik (kecerunan paling curam). FCR berada pada tahap terbaik. Pastikan catuan makanan (TMR) bertenaga tinggi dan dedak diberikan tanpa gangguan untuk mengekalkan momentum ini.`;
            } else {
                phaseVal.innerText = "Finishing / Plateau";
                laymanText.innerHTML = `Kadar pertumbuhan mula mendatar (ADG menyusut ke <strong>${fPrime_x.toFixed(2)} kg/hari</strong>) kerana berat lembu menghampiri had matang genetik ($W_{max}$). Kos makanan berbanding penambahan daging di fasa ini tidak lagi ekonomi. <strong>Tindakan Husbandry:</strong> Sedia untuk dijadualkan bagi jualan atau penyembelihan bagi mengelakkan kerugian FCR.`;
            }
        }

        function initChart() {
            const data = generateChartData();
            const ctx = document.getElementById('growthChart').getContext('2d');
            
            chart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: data.days,
                    datasets: [
                        {
                            label: 'Weight f(x) (kg)',
                            data: data.weights,
                            borderColor: '#818cf8',
                            backgroundColor: 'rgba(129, 140, 248, 0.1)',
                            yAxisID: 'y',
                            tension: 0.3
                        },
                        {
                            label: 'ADG f\'(x) (kg/day x10)',
                            data: data.adgs,
                            borderColor: '#34d399',
                            backgroundColor: 'transparent',
                            yAxisID: 'y1',
                            tension: 0.3
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { labels: { color: '#9ca3af' } }
                    },
                    scales: {
                        x: { grid: { color: '#1e293b' }, ticks: { color: '#9ca3af' } },
                        y: { position: 'left', grid: { color: '#1e293b' }, ticks: { color: '#9ca3af' } },
                        y1: { position: 'right', grid: { drawOnChartArea: false }, ticks: { color: '#9ca3af' } }
                    },
                    onClick: (e) => {
                        const points = chart.getElementsAtEventForMode(e, 'nearest', { intersect: true }, true);
                        if (points.length > 0) {
                            const index = points[0].index;
                            const clickedDay = chart.data.labels[index];
                            daySlider.value = clickedDay;
                            updateMetrics();
                        }
                    }
                }
            });
        }

        function updateChartData() {
            const data = generateChartData();
            chart.data.labels = data.days;
            chart.data.datasets[0].data = data.weights;
            chart.data.datasets[1].data = data.adgs;
            chart.update();
        }

        daySlider.addEventListener('input', updateMetrics);
        wmaxSlider.addEventListener('input', () => { updateMetrics(); updateChartData(); });
        kSlider.addEventListener('input', () => { updateMetrics(); updateChartData(); });
        etaSlider.addEventListener('input', () => { updateMetrics(); updateChartData(); });

        initChart();
        updateMetrics();
    </script>
